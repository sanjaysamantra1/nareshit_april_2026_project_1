{{userName}}

{{ salary }}

{{ salary | currency : 'INR' }}
{{ salary | currency : '€' }}
{{ salary | currency : '¥' }}
{{ salary | currency : 'USD' : 'code' }}
{{ salary | currency : 'USD' : 'symbol' }}

{{cars | slice : 2 }}
{{cars | slice : 2 : 4}}
{{'Angular is a Framework' | slice : 0 : 7}}
{{'Angular is a Framework' | slice : 0 : -4}}
